char* longestCommonPrefix(char** strs, int strsSize) {
    char* s0 = strs[0];
    for (int j = 0; s0[j]; j++) { // 这里的边界条件是第一行字符串是最短的，所以直接返回它本身作为公共前缀
        for (int i = 0; i < strsSize; i++) {
            if (strs[i][j] != s0[j]) { // 不等的情况还包括一种：strs[i][j]等于'\0'代表有一行字符串长度最短，所以以他它为公共前缀
                // s0 可能是只读的，无法直接修改，需要复制一份
                char* s = malloc(j + 1);
                strncpy(s, s0, j);
                s[j] = '\0';
                return s;
            }
        }
    }
    return s0;
}
