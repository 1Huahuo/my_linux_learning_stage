## 122.[[Medium]78. 子集](https://leetcode.cn/problems/subsets/)——==“回溯-子集型”==&”DFS“

```c
// 回溯-子集型
int** subsets(int* nums, int numsSize, int* returnSize, int** returnColumnSizes) {
    int ansSzie = 1 << numsSize; // 答案的规模，2的n次方，其中n为nums数组长度
    int** ans = (int**)malloc(ansSzie * sizeof(int*));
    *returnColumnSizes = (int*)malloc(ansSzie * sizeof(int));
    *returnSize = 0;

    int* tmp = (int*)malloc(numsSize * sizeof(int));
    int tmpSize = 0; // tmp数组中实际加入的nums中元素个数

    void dfs(int i){
        if(i == numsSize){ // 一种可能出现，加入答案
            ans[*returnSize] = (int*)malloc(numsSize * sizeof(int));
            memcpy(ans[*returnSize], tmp, numsSize * sizeof(int));
            (*returnColumnSizes)[(*returnSize)++] = tmpSize; // 注意(*returnColumnSizes)中的括号，[]优先级大于*
            return;
        }
        // 选择将当前nums[i]加入tmp数组
        tmp[tmpSize++] = nums[i];
        dfs(i + 1); // 在选择将当前nums[i]加入tmp数组的基础上，进入下一层递归，即决定下一层nums[i]是否加入tmp数组
        tmp[--tmpSize] = 11; // 11超出本题nums数组元素的值范围，代表没有加入当前nums[i]（仅帮助理解代码，非必须）

        dfs(i + 1); // 在不将nums[i]加入tmp数组的基础上，进入下一层递归
    }

    dfs(0);

    return ans;
}
```

## 123.[[Medium+]131. 分割回文串](https://leetcode.cn/problems/palindrome-partitioning/)——==“回溯-子集型”==&“DFS”

```c
// 回溯-子集型
bool is_palindrome(char* s, int left, int right){
    while(left < right)
        if(s[left++] != s[right--]) // 不要忘记left与right的更新
            return false;
    return true;
} 

char*** partition(char* s, int* returnSize, int** returnColumnSizes) {
    int n = strlen(s);
    int ansSize = 1 << n - 1;
    char*** ans = (char***)malloc(ansSize * sizeof(char**));
    *returnColumnSizes = (int*)malloc(ansSize * sizeof(int));
    *returnSize = 0;

    char** tmp = (char**)malloc(n * sizeof(char*));
    int tmpSize = 0;

    void dfs(int i, int start){ // i代表当前遍历到当前逗号的左边字符的下标索引，start代表目前还未被切割的子串的起始位置
        if(i == n){ // 遍历到最后一个字符时就代表已经没有未遍历的逗号了，结束，并将当前可能加入答案
            ans[*returnSize] = (char**)malloc(tmpSize * sizeof(char*));
            memcpy(ans[*returnSize], tmp, tmpSize * sizeof(char*)); // （对于非字符串最好还是用内存操作函数）
            (*returnColumnSizes)[(*returnSize)++] = tmpSize; // !!![]优先级大于*
            return;
        }

        // 分割，即选择i与i+1之间的逗号
        if(is_palindrome(s, start, i)){ // start是之前传来的，既然选择i与i+1之间的逗号，那就代表截断除了一个子串[start,i]，下一个子串的起始位置将是i+1
            char* substr = (char*)malloc((i+1-start+1) * sizeof(char));
            strncpy(substr, s + start, /*错误原因：sizeof(substr) 中 substr只是一个指针*/ i+1-start); // strncpy(dest, src, sizeof(dest) - 1);
            // 注意这里s的子串并不是一个完整字符串所以strncpy判定复制的字符数是小于子串代表的字符串长度+1的，所以不会有给substr结尾置'\0'的操作，需要我们手动置'\0'
            substr[i+1-start] = '\0';
            tmp[tmpSize++] = substr;
            dfs(i + 1, i + 1); // 考虑 i+1 后面的逗号怎么选,start=i+1 表示下一个子串从 i+1 开始
            tmp[--tmpSize] = NULL; // 回溯：恢复现场
        }

        // 不分割，即不选i和i+1之间的逗号
        if(i < n - 1) // 注意当i = n - 1时即遍历到最后一个字符时必须分割，因为若是不分割则必会丢失一种有可能正确的可能(即子串[start,n-1])，且可能造成一个空方案(即最后一个子串[start=0,n-1]不会被判断即绝不会被加入tmp)，也就是说i = n - 1时不能走不分割这一步即不能以不分割的状态继续向下递归了
            dfs(i + 1, start); // 考虑 i+1 后面的逗号怎么选
        
        // 回溯的最大特点就是:回溯算法的关键在于遍历所有可能的解空间，而选择顺序不影响最终是否遍历完所有解
    }

    dfs(0, 0);

    return ans;
}
```
